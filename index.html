  <script>
    const noBtn = document.getElementById('noBtn');
    const yesBtn = document.getElementById('yesBtn');
    const question = document.getElementById('question');
    const message = document.getElementById('message');
    const extra = document.getElementById('extra');

    // Dodge logic on mouse move
    document.addEventListener('mousemove', (e) => {
      const rect = noBtn.getBoundingClientRect();
      const btnX = rect.left + rect.width / 2;
      const btnY = rect.top + rect.height / 2;
      const dist = Math.hypot(e.clientX - btnX, e.clientY - btnY);

      if (dist < 150) {  // Sensitivity - lower = dodges from farther
        const angle = Math.atan2(e.clientY - btnY, e.clientX - btnX); // Reverse direction to run away
        const moveDist = 250 + Math.random() * 100;
        let newX = btnX + Math.cos(angle) * moveDist;
        let newY = btnY + Math.sin(angle) * moveDist;

        // Keep button on screen
        newX = Math.max(100, Math.min(window.innerWidth - rect.width - 100, newX - rect.width / 2));
        newY = Math.max(100, Math.min(window.innerHeight - rect.height - 100, newY - rect.height / 2));

        noBtn.style.left = `${newX}px`;
        noBtn.style.top = `${newY}px`;
        noBtn.style.transform = 'none'; // Override initial centering
      }
    });

    // Yes button click -> show celebration
    yesBtn.addEventListener('click', () => {
      question.style.display = 'none';
      document.querySelector('.buttons').style.display = 'none';
      message.style.display = 'block';
      extra.style.display = 'block';
      // Optional: Add confetti or sound later if you want to expand
    });
  </script>
